&nbsp;

#### install

```
npm install use-key-debugger
```

&nbsp;

#### usage

```js
import useKeyDebugger from 'use-key-debugger'

function App() {
  const Key = useKeyDebugger()

  return (
    <div>
      <Key />
      <h2>Hit any key to see debugger</h2>
    </div>
  )
}
```
